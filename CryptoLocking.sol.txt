pragma solidity 0.8.34;

contract CryptoLocking {
    struct Deposit {
        uint256 amount;
        uint256 unlockTime;
        bool exists;
    }

    mapping(address => Deposit) public deposits;
    address public owner;

    event Deposited(address indexed user, uint256 amount, uint256 unlockTime);
    event Withdrawn(address indexed user, uint256 amount);

    constructor() {
        owner = msg.sender;
    }

    function deposit(uint256 lockDurationInMinutes) public payable {
        require(msg.value > 0, "Kuch paise toh bhejo!");
        require(lockDurationInMinutes > 0, "Lock duration 0 se zyada honi chahiye!");
        require(!deposits[msg.sender].exists, "Aapka pehle se deposit hai! Pehle withdraw karo!");

        uint256 unlockTime = block.timestamp + (lockDurationInMinutes * 1 minutes);
        deposits[msg.sender] = Deposit({
            amount: msg.value,
            unlockTime: unlockTime,
            exists: true
        });

        emit Deposited(msg.sender, msg.value, unlockTime);
    }

    function withdraw() public {
        require(deposits[msg.sender].exists, "Aapka koi deposit nahi hai!");
        Deposit storage userDeposit = deposits[msg.sender];
        require(block.timestamp >= userDeposit.unlockTime, "Abhi time nahi hua! Wait karo!");
        require(userDeposit.amount > 0, "Deposit amount 0 hai!");

        uint256 amount = userDeposit.amount;
        delete deposits[msg.sender];

        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Paise bhejne mein fail ho gaye!");

        emit Withdrawn(msg.sender, amount);
    }

    function getDepositDetails(address user) public view returns (
        uint256 amount,
        uint256 unlockTime,
        uint256 timeRemaining,
        bool locked
    ) {
        Deposit storage userDeposit = deposits[user];
        if (!userDeposit.exists) {
            return (0, 0, 0, false);
        }

        amount = userDeposit.amount;
        unlockTime = userDeposit.unlockTime;

        if (block.timestamp >= unlockTime) {
            timeRemaining = 0;
            locked = false;
        } else {
            timeRemaining = unlockTime - block.timestamp;
            locked = true;
        }
    }

    function getDepositAmount(address user) public view returns (uint256) {
        return deposits[user].exists ? deposits[user].amount : 0;
    }

    function getUnlockTime(address user) public view returns (uint256) {
        return deposits[user].exists ? deposits[user].unlockTime : 0;
    }

    function getTimeRemaining(address user) public view returns (uint256) {
        if (!deposits[user].exists) {
            return 0;
        }
        uint256 unlockTime = deposits[user].unlockTime;
        return block.timestamp >= unlockTime ? 0 : unlockTime - block.timestamp;
    }

    function isLocked(address user) public view returns (bool) {
        return deposits[user].exists && block.timestamp < deposits[user].unlockTime;
    }

    function getContractBalance() public view returns (uint256) {
        return address(this).balance;
    }

    function emergencyWithdraw() public {
        require(msg.sender == owner, "Sirf owner emergency withdraw kar sakta hai!");
        uint256 balance = address(this).balance;
        require(balance > 0, "Contract mein kuch paise nahi hain!");

        (bool success, ) = owner.call{value: balance}("");
        require(success, "Emergency withdraw fail ho gaya!");
    }
}
